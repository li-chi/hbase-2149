138   /** A call waiting for a value. */
139   private class Call {
140     final int id;                                       // call id
141     final Writable param;                               // parameter
142     Writable value;                               // value, null if error
143     IOException error;                            // exception, null if value
144     boolean done;                                 // true when call is done
145 
146     protected Call(Writable param) {
147       this.param = param;
148       synchronized (HBaseClient.this) {
149         this.id = counter++;
150       }
151     }
152 
153     /** Indicate when the call is complete and the
154      * value or error are available.  Notifies by default.  */
155     protected synchronized void callComplete() {
156       this.done = true;
157       notify();                                 // notify caller
158     }
159 
160     /** Set the exception when there is an error.
161      * Notify the caller the call is done.
162      *
163      * @param error exception thrown by the call; either local or remote
164      */
165     public synchronized void setException(IOException error) {
166       this.error = error;
167       callComplete();
168     }
169 
170     /** Set the return value when there is no error.
171      * Notify the caller the call is done.
172      *
173      * @param value return value of the call.
174      */
175     public synchronized void setValue(Writable value) {
176       this.value = value;
177       callComplete();
178     }

181   /** Thread that reads responses and notifies callers.  Each connection owns a
182    * socket connected to a remote address.  Calls are multiplexed through this
183    * socket: responses may be delivered out of order. */
184   private class Connection extends Thread {
185     private ConnectionId remoteId;
186     private Socket socket = null;                 // connected socket
187     private DataInputStream in;
188     private DataOutputStream out;
189 
190     // currently active calls
191     private final Hashtable<Integer, Call> calls = new Hashtable<Integer, Call>();
192     private final AtomicLong lastActivity = new AtomicLong();// last I/O activity time
193     protected final AtomicBoolean shouldCloseConnection = new AtomicBoolean();  // indicate if the connection is closed
194     private IOException closeException; // close reason
195 
196     public Connection(InetSocketAddress address) throws IOException {
197       this(new ConnectionId(address, null, 0));
198     }

200     public Connection(ConnectionId remoteId) throws IOException {
201       if (remoteId.getAddress().isUnresolved()) {
202         throw new UnknownHostException("unknown host: " +
203                                        remoteId.getAddress().getHostName());
204       }
205       this.remoteId = remoteId;
206       UserGroupInformation ticket = remoteId.getTicket();
207       this.setName("IPC Client (" + socketFactory.hashCode() +") connection to " +
208         remoteId.getAddress().toString() +
209         ((ticket==null)?" from an unknown user": (" from " + ticket.getUserName())));
210       this.setDaemon(true);
211     }

213     /** Update lastActivity with the current time. */
214     private void touch() {
215       lastActivity.set(System.currentTimeMillis());
216     }

218     /**
219      * Add a call to this connection's call queue and notify
220      * a listener; synchronized.
221      * Returns false if called during shutdown.
222      * @param call to add
223      * @return true if the call was added.
224      */
225     protected synchronized boolean addCall(Call call) {
226       if (shouldCloseConnection.get())
227         return false;
228       calls.put(call.id, call);
229       notify();
230       return true;
231     }

233     /** This class sends a ping to the remote side when timeout on
234      * reading. If no failure is detected, it retries until at least
235      * a byte is read.
236      */
237     private class PingInputStream extends FilterInputStream {
238       /* constructor */
239       protected PingInputStream(InputStream in) {
240         super(in);
241       }
242 
243       /* Process timeout exception
244        * if the connection is not going to be closed, send a ping.
245        * otherwise, throw the timeout exception.
246        */
247       private void handleTimeout(SocketTimeoutException e) throws IOException {
248         if (shouldCloseConnection.get() || !running.get() || 
249             remoteId.rpcTimeout > 0) {
250           throw e;
251         }
252         sendPing();
253       }
254 
255       /** Read a byte from the stream.
256        * Send a ping if timeout on read. Retries if no failure is detected
257        * until a byte is read.
258        * @throws IOException for any IO problem other than socket timeout
259        */
260       @Override
261       public int read() throws IOException {
262         do {
263           try {
264             return super.read();
265           } catch (SocketTimeoutException e) {
266             handleTimeout(e);
267           }
268         } while (true);
269       }
270 
271       /** Read bytes into a buffer starting from offset <code>off</code>
272        * Send a ping if timeout on read. Retries if no failure is detected
273        * until a byte is read.
274        *
275        * @return the total number of bytes read; -1 if the connection is closed.
276        */
277       @Override
278       public int read(byte[] buf, int off, int len) throws IOException {
279         do {
280           try {
281             return super.read(buf, off, len);
282           } catch (SocketTimeoutException e) {
283             handleTimeout(e);
284           }
285         } while (true);
286       }
287     }

289     /** Connect to the server and set up the I/O streams. It then sends
290      * a header to the server and starts
291      * the connection thread that waits for responses.
292      * @throws java.io.IOException e
293      */
294     protected synchronized void setupIOstreams() throws IOException {
295       if (socket != null || shouldCloseConnection.get()) {
296         return;
297       }
298 
299       short ioFailures = 0;
300       short timeoutFailures = 0;
301       try {
302         if (LOG.isDebugEnabled()) {
303           LOG.debug("Connecting to "+remoteId.getAddress());
304         }
305         while (true) {
306           try {
307             this.socket = socketFactory.createSocket();
308             this.socket.setTcpNoDelay(tcpNoDelay);
309             this.socket.setKeepAlive(tcpKeepAlive);
310             // connection time out is 20s
311             NetUtils.connect(this.socket, remoteId.getAddress(), 20000);
312             if (remoteId.rpcTimeout > 0) {
313               pingInterval = remoteId.rpcTimeout; // overwrite pingInterval
314             }
315             this.socket.setSoTimeout(pingInterval);
316             break;
317           } catch (SocketTimeoutException toe) {
318             handleConnectionFailure(timeoutFailures++, maxRetries, toe);
319           } catch (IOException ie) {
320             handleConnectionFailure(ioFailures++, maxRetries, ie);
321           }
322         }
323         this.in = new DataInputStream(new BufferedInputStream
324             (new PingInputStream(NetUtils.getInputStream(socket))));
325         this.out = new DataOutputStream
326             (new BufferedOutputStream(NetUtils.getOutputStream(socket)));
327         writeHeader();
328 
329         // update last activity time
330         touch();
331 
332         // start the receiver thread after the socket connection has been set up
333         start();
334       } catch (IOException e) {
335         markClosed(e);
336         close();
337 
338         throw e;
339       }
340     }
341 
342     /* Handle connection failures
343      *
344      * If the current number of retries is equal to the max number of retries,
345      * stop retrying and throw the exception; Otherwise backoff N seconds and
346      * try connecting again.
347      *
348      * This Method is only called from inside setupIOstreams(), which is
349      * synchronized. Hence the sleep is synchronized; the locks will be retained.
350      *
351      * @param curRetries current number of retries
352      * @param maxRetries max number of retries allowed
353      * @param ioe failure reason
354      * @throws IOException if max number of retries is reached
355      */
356     private void handleConnectionFailure(
357         int curRetries, int maxRetries, IOException ioe) throws IOException {
358       // close the current connection
359       if (socket != null) { // could be null if the socket creation failed
360         try {
361           socket.close();
362         } catch (IOException e) {
363           LOG.warn("Not able to close a socket", e);
364         }
365       }
366       // set socket to null so that the next call to setupIOstreams
367       // can start the process of connect all over again.
368       socket = null;
369 
370       // throw the exception if the maximum number of retries is reached
371       if (curRetries >= maxRetries) {
372         throw ioe;
373       }
374 
375       // otherwise back off and retry
376       try {
377         Thread.sleep(failureSleep);
378       } catch (InterruptedException ignored) {}
379 
380       LOG.info("Retrying connect to server: " + remoteId.getAddress() +
381         " after sleeping " + failureSleep + "ms. Already tried " + curRetries +
382         " time(s).");
383     }

385     /* Write the header for each connection
386      * Out is not synchronized because only the first thread does this.
387      */
388     private void writeHeader() throws IOException {
389       out.write(HBaseServer.HEADER.array());
390       out.write(HBaseServer.CURRENT_VERSION);
391       //When there are more fields we can have ConnectionHeader Writable.
392       DataOutputBuffer buf = new DataOutputBuffer();
393       ObjectWritable.writeObject(buf, remoteId.getTicket(),
394                                  UserGroupInformation.class, conf);
395       int bufLen = buf.getLength();
396       out.writeInt(bufLen);
397       out.write(buf.getData(), 0, bufLen);
398     }

400     /* wait till someone signals us to start reading RPC response or
401      * it is idle too long, it is marked as to be closed,
402      * or the client is marked as not running.
403      *
404      * Return true if it is time to read a response; false otherwise.
405      */
406     @SuppressWarnings({"ThrowableInstanceNeverThrown"})
407     private synchronized boolean waitForWork() {
408       if (calls.isEmpty() && !shouldCloseConnection.get()  && running.get())  {
409         long timeout = maxIdleTime-
410               (System.currentTimeMillis()-lastActivity.get());
411         if (timeout>0) {
412           try {
413             wait(timeout);
414           } catch (InterruptedException ignored) {}
415         }
416       }
417 
418       if (!calls.isEmpty() && !shouldCloseConnection.get() && running.get()) {
419         return true;
420       } else if (shouldCloseConnection.get()) {
421         return false;
422       } else if (calls.isEmpty()) { // idle connection closed or stopped
423         markClosed(null);
424         return false;
425       } else { // get stopped but there are still pending requests
426         markClosed((IOException)new IOException().initCause(
427             new InterruptedException()));
428         return false;
429       }
430     }

436     /* Send a ping to the server if the time elapsed
437      * since last I/O activity is equal to or greater than the ping interval
438      */
439     protected synchronized void sendPing() throws IOException {
440       long curTime = System.currentTimeMillis();
441       if ( curTime - lastActivity.get() >= pingInterval) {
442         lastActivity.set(curTime);
443         //noinspection SynchronizeOnNonFinalField
444         synchronized (this.out) {
445           out.writeInt(PING_CALL_ID);
446           out.flush();
447         }
448       }
449     }

451     @Override
452     public void run() {
453       if (LOG.isDebugEnabled())
454         LOG.debug(getName() + ": starting, having connections "
455             + connections.size());
456 
457       try {
458         while (waitForWork()) {//wait here for work - read or close connection
459           receiveResponse();
460         }
461       } catch (Throwable t) {
462         LOG.warn("Unexpected exception receiving call responses", t);
463         markClosed(new IOException("Unexpected exception receiving call responses", t));
464       }
465 
466       close();
467 
468       if (LOG.isDebugEnabled())
469         LOG.debug(getName() + ": stopped, remaining connections "
470             + connections.size());
471     }

473     /* Initiates a call by sending the parameter to the remote server.
474      * Note: this is not called from the Connection thread, but by other
475      * threads.
476      */
477     protected void sendParam(Call call) {
478       if (shouldCloseConnection.get()) {
479         return;
480       }
481 
482       DataOutputBuffer d=null;
483       try {
484         //noinspection SynchronizeOnNonFinalField
485         synchronized (this.out) { // FindBugs IS2_INCONSISTENT_SYNC
486           if (LOG.isDebugEnabled())
487             LOG.debug(getName() + " sending #" + call.id);
488 
489           //for serializing the
490           //data to be written
491           d = new DataOutputBuffer();
492           d.writeInt(0xdeadbeef); // placeholder for data length
493           d.writeInt(call.id);
494           call.param.write(d);
495           byte[] data = d.getData();
496           int dataLength = d.getLength();
497           // fill in the placeholder
498           Bytes.putInt(data, 0, dataLength - 4);
499           out.write(data, 0, dataLength);
500           out.flush();
501         }
502       } catch(IOException e) {
503         markClosed(e);
504       } finally {
505         //the buffer is just an in-memory buffer, but it is still polite to
506         // close early
507         IOUtils.closeStream(d);
508       }
509     }
510 
511     /* Receive a response.
512      * Because only one receiver, so no synchronization on in.
513      */
514     private void receiveResponse() {
515       if (shouldCloseConnection.get()) {
516         return;
517       }
518       touch();
519 
520       try {
521         int id = in.readInt();                    // try to read an id
522 
523         if (LOG.isDebugEnabled())
524           LOG.debug(getName() + " got value #" + id);
525 
526         Call call = calls.get(id);
527 
528         boolean isError = in.readBoolean();     // read if error
529         if (isError) {
530           //noinspection ThrowableInstanceNeverThrown
531           call.setException(new RemoteException( WritableUtils.readString(in),
532               WritableUtils.readString(in)));
533           calls.remove(id);
534         } else {
535           Writable value = ReflectionUtils.newInstance(valueClass, conf);
536           value.readFields(in);                 // read value
537           call.setValue(value);
538           calls.remove(id);
539         }
540       } catch (IOException e) {
541         markClosed(e);
542       }
543     }
544 
545     private synchronized void markClosed(IOException e) {
546       if (shouldCloseConnection.compareAndSet(false, true)) {
547         closeException = e;
548         notifyAll();
549       }
550     }
551 
552     /** Close the connection. */
553     private synchronized void close() {
554       if (!shouldCloseConnection.get()) {
555         LOG.error("The connection is not in the closed state");
556         return;
557       }
558 
559       // release the resources
560       // first thing to do;take the connection out of the connection list
561       synchronized (connections) {
562         if (connections.get(remoteId) == this) {
563           connections.remove(remoteId);
564         }
565       }
566 
567       // close the streams and therefore the socket
568       IOUtils.closeStream(out);
569       IOUtils.closeStream(in);
570 
571       // clean up all calls
572       if (closeException == null) {
573         if (!calls.isEmpty()) {
574           LOG.warn(
575               "A connection is closed for no cause and calls are not empty");
576 
577           // clean up calls anyway
578           closeException = new IOException("Unexpected closed connection");
579           cleanupCalls();
580         }
581       } else {
582         // log the info
583         if (LOG.isDebugEnabled()) {
584           LOG.debug("closing ipc connection to " + remoteId.address + ": " +
585               closeException.getMessage(),closeException);
586         }
587 
588         // cleanup calls
589         cleanupCalls();
590       }
591       if (LOG.isDebugEnabled())
592         LOG.debug(getName() + ": closed");
593     }
594 
595     /* Cleanup all calls and mark them as done */
596     private void cleanupCalls() {
597       Iterator<Entry<Integer, Call>> itor = calls.entrySet().iterator() ;
598       while (itor.hasNext()) {
599         Call c = itor.next().getValue();
600         c.setException(closeException); // local exception
601         itor.remove();
602       }
603     }
604   }

624   /** Result collector for parallel calls. */
625   private static class ParallelResults {
626     protected final Writable[] values;
627     protected int size;
628     protected int count;
629 
630     public ParallelResults(int size) {
631       this.values = new Writable[size];
632       this.size = size;
633     }
634 
635     /*
636      * Collect a result.
637      */
638     synchronized void callComplete(ParallelCall call) {
639       // FindBugs IS2_INCONSISTENT_SYNC
640       values[call.index] = call.value;            // store the value
641       count++;                                    // count it
642       if (count == size)                          // if all values are in
643         notify();                                 // then notify waiting caller
644     }
645   }

688   /** Stop all threads related to this client.  No further calls may be made
689    * using this client. */
690   public void stop() {
691     if (LOG.isDebugEnabled()) {
692       LOG.debug("Stopping client");
693     }
694 
695     if (!running.compareAndSet(true, false)) {
696       return;
697     }
698 
699     // wake up all connections
700     synchronized (connections) {
701       for (Connection conn : connections.values()) {
702         conn.interrupt();
703       }
704     }
705 
706     // wait until all connections are closed
707     while (!connections.isEmpty()) {
708       try {
709         Thread.sleep(100);
710       } catch (InterruptedException ignored) {
711       }
712     }
713   }

715   /** Make a call, passing <code>param</code>, to the IPC server running at
716    * <code>address</code>, returning the value.  Throws exceptions if there are
717    * network problems or if the remote code threw an exception.
718    * @param param writable parameter
719    * @param address network address
720    * @return Writable
721    * @throws IOException e
722    */
723   public Writable call(Writable param, InetSocketAddress address)
724   throws IOException {
725       return call(param, address, null, 0);
726   }

728   public Writable call(Writable param, InetSocketAddress addr,
729                        UserGroupInformation ticket, int rpcTimeout)
730                        throws IOException {
731     Call call = new Call(param);
732     Connection connection = getConnection(addr, ticket, rpcTimeout, call);
733     connection.sendParam(call);                 // send the parameter
734     boolean interrupted = false;
735     //noinspection SynchronizationOnLocalVariableOrMethodParameter
736     synchronized (call) {
737       while (!call.done) {
738         try {
739           call.wait();                           // wait for the result
740         } catch (InterruptedException ignored) {
741           // save the fact that we were interrupted
742           interrupted = true;
743         }
744       }
745 
746       if (interrupted) {
747         // set the interrupt flag now that we are done waiting
748         Thread.currentThread().interrupt();
749       }
750 
751       if (call.error != null) {
752         if (call.error instanceof RemoteException) {
753           call.error.fillInStackTrace();
754           throw call.error;
755         }
756         // local exception
757         throw wrapException(addr, call.error);
758       }
759       return call.value;
760     }

777                                          IOException exception) {
778     if (exception instanceof ConnectException) {

780       return (ConnectException)new ConnectException(

795   /** Makes a set of calls in parallel.  Each parameter is sent to the
796    * corresponding address.  When all values are available, or have timed out
797    * or errored, the collected results are returned in an array.  The array
798    * contains nulls for calls that timed out or errored.
799    * @param params writable parameters
800    * @param addresses socket addresses
801    * @return  Writable[]
802    * @throws IOException e
803    */
804   public Writable[] call(Writable[] params, InetSocketAddress[] addresses)
805     throws IOException {
806     if (addresses.length == 0) return new Writable[0];
807 
808     ParallelResults results = new ParallelResults(params.length);
809     // TODO this synchronization block doesnt make any sense, we should possibly fix it
810     //noinspection SynchronizationOnLocalVariableOrMethodParameter
811     synchronized (results) {
812       for (int i = 0; i < params.length; i++) {
813         ParallelCall call = new ParallelCall(params[i], results, i);
814         try {
815           Connection connection = getConnection(addresses[i], null, 0, call);
816           connection.sendParam(call);             // send each parameter
817         } catch (IOException e) {
818           // log errors
819           LOG.info("Calling "+addresses[i]+" caught: " +
820                    e.getMessage(),e);
821           results.size--;                         //  wait for one fewer result
822         }
823       }
824       while (results.count != results.size) {
825         try {
826           results.wait();                    // wait for all results
827         } catch (InterruptedException ignored) {}
828       }
829 
830       return results.values;
831     }
832   }

834   /* Get a connection from the pool, or create a new one and add it to the
835    * pool.  Connections to a given host/port are reused. */
836   private Connection getConnection(InetSocketAddress addr,
837                                    UserGroupInformation ticket,
838                                    int rpcTimeout,
839                                    Call call)
840                                    throws IOException {
841     if (!running.get()) {
842       // the client is stopped
843       throw new IOException("The client is stopped");
844     }
845     Connection connection;
846     /* we could avoid this allocation for each RPC by having a
847      * connectionsId object and with set() method. We need to manage the
848      * refs for keys in HashMap properly. For now its ok.
849      */
850     ConnectionId remoteId = new ConnectionId(addr, ticket, rpcTimeout);
851     do {
852       synchronized (connections) {
853         connection = connections.get(remoteId);
854         if (connection == null) {
855           connection = new Connection(remoteId);
856           connections.put(remoteId, connection);
857         }
858       }
859     } while (!connection.addCall(call));
860 
861     //we don't invoke the method below inside "synchronized (connections)"
862     //block above. The reason for that is if the server happens to be slow,
863     //it will take longer to establish a connection and that will slow the
864     //entire system down.
865     connection.setupIOstreams();
866     return connection;
867   }
