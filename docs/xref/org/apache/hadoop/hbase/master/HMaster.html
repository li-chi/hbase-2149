93  /**
94   * HMaster is the "master server" for HBase. An HBase cluster has one active
95   * master.  If many masters are started, all compete.  Whichever wins goes on to
96   * run the cluster.  All others park themselves in their constructor until
97   * master or cluster shutdown or until the active master loses its lease in
98   * zookeeper.  Thereafter, all running master jostle to take over master role.
99   *
100  * <p>The Master can be asked shutdown the cluster. See {@link #shutdown()}.  In
101  * this case it will tell all regionservers to go down and then wait on them
102  * all reporting in that they are down.  This master will then shut itself down.
103  *
104  * <p>You can also shutdown just this master.  Call {@link #stopMaster()}.
105  *
106  * @see HMasterInterface
107  * @see HMasterRegionInterface
108  * @see Watcher
109  */
110 public class HMaster extends Thread
111 implements HMasterInterface, HMasterRegionInterface, MasterServices, Server {

172   /**
173    * Initializes the HMaster. The steps are as follows:
174    * <p>
175    * <ol>
176    * <li>Initialize HMaster RPC and address
177    * <li>Connect to ZooKeeper.
178    * </ol>
179    * <p>
180    * Remaining steps of initialization occur in {@link #run()} so that they
181    * run in their own thread rather than within the context of the constructor.
182    * @throws InterruptedException
183    */
184   public HMaster(final Configuration conf)
185   throws IOException, KeeperException, InterruptedException {
186     this.conf = conf;
187 
188     /*
189      * Determine address and initialize RPC server (but do not start).
190      * The RPC server ports can be ephemeral. Create a ZKW instance.
191      */
192     HServerAddress a = new HServerAddress(getMyAddress(this.conf));
193     int numHandlers = conf.getInt("hbase.regionserver.handler.count", 10);
194     this.rpcServer = HBaseRPC.getServer(this,
195       new Class<?>[]{HMasterInterface.class, HMasterRegionInterface.class},
196       a.getBindAddress(), a.getPort(),
197       numHandlers,
198       0, // we dont use high priority handlers in master
199       false, conf,
200       0); // this is a DNC w/o high priority handlers
201     this.address = new HServerAddress(rpcServer.getListenerAddress());
202 
203     // set the thread name now we have an address
204     setName(MASTER + "-" + this.address);
205 
206     Replication.decorateMasterConfiguration(this.conf);
207 
208     this.rpcServer.startThreads();
209 
210     // Hack! Maps DFSClient => Master for logs.  HDFS made this
211     // config param for task trackers, but we can piggyback off of it.
212     if (this.conf.get("mapred.task.id") == null) {
213       this.conf.set("mapred.task.id", "hb_m_" + this.address.toString() +
214         "_" + System.currentTimeMillis());
215     }
216 
217     this.zooKeeper = new ZooKeeperWatcher(conf, MASTER + ":" +
218         address.getPort(), this);
219 
220     this.metrics = new MasterMetrics(getServerName());
221   }

223   /**
224    * Stall startup if we are designated a backup master; i.e. we want someone
225    * else to become the master before proceeding.
226    * @param c
227    * @param amm
228    * @throws InterruptedException
229    */
230   private static void stallIfBackupMaster(final Configuration c,
231       final ActiveMasterManager amm)
232   throws InterruptedException {
233     // If we're a backup master, stall until a primary to writes his address
234     if (!c.getBoolean(HConstants.MASTER_TYPE_BACKUP,
235       HConstants.DEFAULT_MASTER_TYPE_BACKUP)) {
236       return;
237     }
238     LOG.debug("HMaster started in backup mode.  " +
239       "Stalling until master znode is written.");
240     // This will only be a minute or so while the cluster starts up,
241     // so don't worry about setting watches on the parent znode
242     while (!amm.isActiveMaster()) {
243       LOG.debug("Waiting for master address ZNode to be written " +
244         "(Also watching cluster state node)");
245       Thread.sleep(c.getInt("zookeeper.session.timeout", 180 * 1000));
246     }
247   }

249   /**
250    * Main processing loop for the HMaster.
251    * <ol>
252    * <li>Block until becoming active master
253    * <li>Finish initialization via {@link #finishInitialization()}
254    * <li>Enter loop until we are stopped
255    * <li>Stop services and perform cleanup once stopped
256    * </ol>
257    */
258   @Override
259   public void run() {
260     try {
261       /*
262        * Block on becoming the active master.
263        *
264        * We race with other masters to write our address into ZooKeeper.  If we
265        * succeed, we are the primary/active master and finish initialization.
266        *
267        * If we do not succeed, there is another active master and we should
268        * now wait until it dies to try and become the next active master.  If we
269        * do not succeed on our first attempt, this is no longer a cluster startup.
270        */
271       this.activeMasterManager = new ActiveMasterManager(zooKeeper, address, this);
272       this.zooKeeper.registerListener(activeMasterManager);
273       stallIfBackupMaster(this.conf, this.activeMasterManager);
274       this.activeMasterManager.blockUntilBecomingActiveMaster();
275       // We are either the active master or we were asked to shutdown
276       if (!this.stopped) {
277         finishInitialization();
278         loop();
279       }
280     } catch (Throwable t) {
281       abort("Unhandled exception. Starting shutdown.", t);
282     } finally {
283       stopChores();
284       // Wait for all the remaining region servers to report in IFF we were
285       // running a cluster shutdown AND we were NOT aborting.
286       if (!this.abort && this.serverManager != null &&
287           this.serverManager.isClusterShutdown()) {
288         this.serverManager.letRegionServersShutdown();
289       }
290       stopServiceThreads();
291       // Stop services started for both backup and active masters
292       if (this.activeMasterManager != null) this.activeMasterManager.stop();
293       if (this.catalogTracker != null) this.catalogTracker.stop();
294       if (this.serverManager != null) this.serverManager.stop();
295       if (this.assignmentManager != null) this.assignmentManager.stop();
296       HConnectionManager.deleteConnection(this.conf, true);
297       this.zooKeeper.close();
298     }
299     LOG.info("HMaster main thread exiting");
300   }
301 
302   private void loop() {
303     // Check if we should stop every second.
304     Sleeper sleeper = new Sleeper(1000, this);
305     while (!this.stopped) {
306       sleeper.sleep();
307     }
308   }

310   /**
311    * Finish initialization of HMaster after becoming the primary master.
312    *
313    * <ol>
314    * <li>Initialize master components - file system manager, server manager,
315    *     assignment manager, region server tracker, catalog tracker, etc</li>
316    * <li>Start necessary service threads - rpc server, info server,
317    *     executor services, etc</li>
318    * <li>Set cluster as UP in ZooKeeper</li>
319    * <li>Wait for RegionServers to check-in</li>
320    * <li>Split logs and perform data recovery, if necessary</li>
321    * <li>Ensure assignment of root and meta regions<li>
322    * <li>Handle either fresh cluster start or master failover</li>
323    * </ol>
324    *
325    * @throws IOException
326    * @throws InterruptedException
327    * @throws KeeperException
328    */
329   private void finishInitialization()
330   throws IOException, InterruptedException, KeeperException {
331 
332     isActiveMaster = true;
333 
334     /*
335      * We are active master now... go initialize components we need to run.
336      * Note, there may be dross in zk from previous runs; it'll get addressed
337      * below after we determine if cluster startup or failover.
338      */
339 
340     // TODO: Do this using Dependency Injection, using PicoContainer, Guice or Spring.
341     this.fileSystemManager = new MasterFileSystem(this, metrics);
342     this.connection = HConnectionManager.getConnection(conf);
343     this.executorService = new ExecutorService(getServerName());
344 
345     this.serverManager = new ServerManager(this, this, metrics);
346 
347     this.catalogTracker = new CatalogTracker(this.zooKeeper, this.connection,
348       this, conf.getInt("hbase.master.catalog.timeout", Integer.MAX_VALUE));
349     this.catalogTracker.start();
350 
351     this.assignmentManager = new AssignmentManager(this, serverManager,
352       this.catalogTracker, this.executorService);
353     zooKeeper.registerListenerFirst(assignmentManager);
354 
355     this.regionServerTracker = new RegionServerTracker(zooKeeper, this,
356       this.serverManager);
357     this.regionServerTracker.start();
358 
359     // Set the cluster as up.  If new RSs, they'll be waiting on this before
360     // going ahead with their startup.
361     this.clusterStatusTracker = new ClusterStatusTracker(getZooKeeper(), this);
362     this.clusterStatusTracker.start();
363     boolean wasUp = this.clusterStatusTracker.isClusterUp();
364     if (!wasUp) this.clusterStatusTracker.setClusterUp();
365 
366     LOG.info("Server active/primary master; " + this.address +
367       ", sessionid=0x" +
368       Long.toHexString(this.zooKeeper.getZooKeeper().getSessionId()) +
369       ", cluster-up flag was=" + wasUp);
370 
371     // start up all service threads.
372     startServiceThreads();
373 
374     // Wait for region servers to report in.  Returns count of regions.
375     int regionCount = this.serverManager.waitForRegionServers();
376 
377     // TODO: Should do this in background rather than block master startup
378     this.fileSystemManager.
379       splitLogAfterStartup(this.serverManager.getOnlineServers());
380 
381     // Make sure root and meta assigned before proceeding.
382     assignRootAndMeta();
383 
384     // Is this fresh start with no regions assigned or are we a master joining
385     // an already-running cluster?  If regionsCount == 0, then for sure a
386     // fresh start.  TOOD: Be fancier.  If regionsCount == 2, perhaps the
387     // 2 are .META. and -ROOT- and we should fall into the fresh startup
388     // branch below.  For now, do processFailover.
389     if (regionCount == 0) {
390       LOG.info("Master startup proceeding: cluster startup");
391       this.assignmentManager.cleanoutUnassigned();
392       this.assignmentManager.assignAllUserRegions();
393     } else {
394       LOG.info("Master startup proceeding: master failover");
395       this.assignmentManager.processFailover();
396     }
397 
398     // Start balancer and meta catalog janitor after meta and regions have
399     // been assigned.
400     this.balancerChore = getAndStartBalancerChore(this);
401     this.catalogJanitorChore =
402       Threads.setDaemonThreadRunning(new CatalogJanitor(this, this));
403 
404     LOG.info("Master has completed initialization");
405     initialized = true;
406   }

408   /**
409    * Check <code>-ROOT-</code> and <code>.META.</code> are assigned.  If not,
410    * assign them.
411    * @throws InterruptedException
412    * @throws IOException
413    * @throws KeeperException
414    * @return Count of regions we assigned.
415    */
416   int assignRootAndMeta()
417   throws InterruptedException, IOException, KeeperException {
418     int assigned = 0;
419     long timeout = this.conf.getLong("hbase.catalog.verification.timeout", 1000);
420 
421     // Work on ROOT region.  Is it in zk in transition?
422     boolean rit = this.assignmentManager.
423       processRegionInTransitionAndBlockUntilAssigned(HRegionInfo.ROOT_REGIONINFO);
424     if (!catalogTracker.verifyRootRegionLocation(timeout)) {
425       this.assignmentManager.assignRoot();
426       this.catalogTracker.waitForRoot();
427       assigned++;
428     }
429     LOG.info("-ROOT- assigned=" + assigned + ", rit=" + rit +
430       ", location=" + catalogTracker.getRootLocation());
431 
432     // Work on meta region
433     rit = this.assignmentManager.
434       processRegionInTransitionAndBlockUntilAssigned(HRegionInfo.FIRST_META_REGIONINFO);
435     if (!this.catalogTracker.verifyMetaRegionLocation(timeout)) {
436       this.assignmentManager.assignMeta();
437       this.catalogTracker.waitForMeta();
438       // Above check waits for general meta availability but this does not
439       // guarantee that the transition has completed
440       this.assignmentManager.waitForAssignment(HRegionInfo.FIRST_META_REGIONINFO);
441       assigned++;
442     }
443     LOG.info(".META. assigned=" + assigned + ", rit=" + rit +
444       ", location=" + catalogTracker.getMetaLocation());
445     return assigned;
446   }

504   /*
505    * Start up all services. If any of these threads gets an unhandled exception
506    * then they just die with a logged message.  This should be fine because
507    * in general, we do not expect the master to get such unhandled exceptions
508    *  as OOMEs; it should be lightly loaded. See what HRegionServer does if
509    *  need to install an unexpected exception handler.
510    */
511   private void startServiceThreads() {
512     try {
513       // Start the executor service pools
514       this.executorService.startExecutorService(ExecutorType.MASTER_OPEN_REGION,
515         conf.getInt("hbase.master.executor.openregion.threads", 5));
516       this.executorService.startExecutorService(ExecutorType.MASTER_CLOSE_REGION,
517         conf.getInt("hbase.master.executor.closeregion.threads", 5));
518       this.executorService.startExecutorService(ExecutorType.MASTER_SERVER_OPERATIONS,
519         conf.getInt("hbase.master.executor.serverops.threads", 3));
520       this.executorService.startExecutorService(ExecutorType.MASTER_META_SERVER_OPERATIONS,
521         conf.getInt("hbase.master.executor.serverops.threads", 2));
522       // We depend on there being only one instance of this executor running
523       // at a time.  To do concurrency, would need fencing of enable/disable of
524       // tables.
525       this.executorService.startExecutorService(ExecutorType.MASTER_TABLE_OPERATIONS, 1);
526 
527       // Start log cleaner thread
528       String n = Thread.currentThread().getName();
529       this.logCleaner =
530         new LogCleaner(conf.getInt("hbase.master.cleaner.interval", 60 * 1000),
531           this, conf, getMasterFileSystem().getFileSystem(),
532           getMasterFileSystem().getOldLogDir());
533       Threads.setDaemonThreadRunning(logCleaner, n + ".oldLogCleaner");
534 
535       // Put up info server.
536       int port = this.conf.getInt("hbase.master.info.port", 60010);
537       if (port >= 0) {
538         String a = this.conf.get("hbase.master.info.bindAddress", "0.0.0.0");
539         this.infoServer = new InfoServer(MASTER, a, port, false);
540         this.infoServer.setAttribute(MASTER, this);
541         this.infoServer.start();
542       }
543       // Start allowing requests to happen.
544       this.rpcServer.openServer();
545       if (LOG.isDebugEnabled()) {
546         LOG.debug("Started service threads");
547       }
548     } catch (IOException e) {
549       if (e instanceof RemoteException) {
550         e = ((RemoteException)e).unwrapRemoteException();
551       }
552       // Something happened during startup. Shut things down.
553       abort("Failed startup", e);
554     }
555   }
556 
557   private void stopServiceThreads() {
558     if (LOG.isDebugEnabled()) {
559       LOG.debug("Stopping service threads");
560     }
561     if (this.rpcServer != null) this.rpcServer.stop();
562     // Clean up and close up shop
563     if (this.logCleaner!= null) this.logCleaner.interrupt();
564     if (this.infoServer != null) {
565       LOG.info("Stopping infoServer");
566       try {
567         this.infoServer.stop();
568       } catch (Exception ex) {
569         ex.printStackTrace();
570       }
571     }
572     if (this.executorService != null) this.executorService.shutdown();
573   }

575   private static Thread getAndStartBalancerChore(final HMaster master) {
576     String name = master.getServerName() + "-BalancerChore";
577     int period = master.getConfiguration().getInt("hbase.balancer.period", 300000);
578     // Start up the load balancer chore
579     Chore chore = new Chore(name, period, master) {
580       @Override
581       protected void chore() {
582         master.balance();
583       }
584     };
585     return Threads.setDaemonThreadRunning(chore);
586   }

597   @Override
598   public MapWritable regionServerStartup(final HServerInfo serverInfo,
599     final long serverCurrentTime)
600   throws IOException {
601     // Set the ip into the passed in serverInfo.  Its ip is more than likely
602     // not the ip that the master sees here.  See at end of this method where
603     // we pass it back to the regionserver by setting "hbase.regionserver.address"
604     // Everafter, the HSI combination 'server name' is what uniquely identifies
605     // the incoming RegionServer.
606     InetSocketAddress address = new InetSocketAddress(
607         HBaseServer.getRemoteIp().getHostName(),
608         serverInfo.getServerAddress().getPort());
609     serverInfo.setServerAddress(new HServerAddress(address));
610 
611     // Register with server manager
612     this.serverManager.regionServerStartup(serverInfo, serverCurrentTime);
613     // Send back some config info
614     MapWritable mw = createConfigurationSubset();
615      mw.put(new Text("hbase.regionserver.address"),
616          serverInfo.getServerAddress());
617     return mw;
618   }

658   @Override
659   public boolean balance() {
660     // If balance not true, don't run balancer.
661     if (!this.balanceSwitch) return false;
662     synchronized (this.balancer) {
663       // Only allow one balance run at at time.
664       if (this.assignmentManager.isRegionsInTransition()) {
665         LOG.debug("Not running balancer because " +
666           this.assignmentManager.getRegionsInTransition().size() +
667           " region(s) in transition: " +
668           org.apache.commons.lang.StringUtils.
669             abbreviate(this.assignmentManager.getRegionsInTransition().toString(), 256));
670         return false;
671       }
672       if (this.serverManager.areDeadServersInProgress()) {
673         LOG.debug("Not running balancer because processing dead regionserver(s): "  +
674           this.serverManager.getDeadServers());
675         return false;
676       }
677       Map<HServerInfo, List<HRegionInfo>> assignments =
678         this.assignmentManager.getAssignments();
679       // Returned Map from AM does not include mention of servers w/o assignments.
680       for (Map.Entry<String, HServerInfo> e:
681           this.serverManager.getOnlineServers().entrySet()) {
682         HServerInfo hsi = e.getValue();
683         if (!assignments.containsKey(hsi)) {
684           assignments.put(hsi, new ArrayList<HRegionInfo>());
685         }
686       }
687       List<RegionPlan> plans = this.balancer.balanceCluster(assignments);
688       if (plans != null && !plans.isEmpty()) {
689         for (RegionPlan plan: plans) {
690           LOG.info("balance " + plan);
691           this.assignmentManager.balance(plan);
692         }
693       }
694     }
695     return true;
696   }

743   public void createTable(HTableDescriptor desc, byte [][] splitKeys,
744       boolean sync)
745   throws IOException {
746     if (!isMasterRunning()) {
747       throw new MasterNotRunningException();
748     }
749     HRegionInfo [] newRegions = null;
750     if(splitKeys == null || splitKeys.length == 0) {
751       newRegions = new HRegionInfo [] { new HRegionInfo(desc, null, null) };
752     } else {
753       int numRegions = splitKeys.length + 1;
754       newRegions = new HRegionInfo[numRegions];
755       byte [] startKey = null;
756       byte [] endKey = null;
757       for(int i=0;i<numRegions;i++) {
758         endKey = (i == splitKeys.length) ? null : splitKeys[i];
759         newRegions[i] = new HRegionInfo(desc, startKey, endKey);
760         startKey = endKey;
761       }
762     }
763     int timeout = conf.getInt("hbase.client.catalog.timeout", 10000);
764     // Need META availability to create a table
765     try {
766       if(catalogTracker.waitForMeta(timeout) == null) {
767         throw new NotAllMetaRegionsOnlineException();
768       }
769     } catch (InterruptedException e) {
770       LOG.warn("Interrupted waiting for meta availability", e);
771       throw new IOException(e);
772     }
773     createTable(newRegions, sync);
774   }

776   private synchronized void createTable(final HRegionInfo [] newRegions,
777       boolean sync)
778   throws IOException {
779     String tableName = newRegions[0].getTableDesc().getNameAsString();
780     if(MetaReader.tableExists(catalogTracker, tableName)) {
781       throw new TableExistsException(tableName);
782     }
783     for(HRegionInfo newRegion : newRegions) {
784 
785       // 1. Set table enabling flag up in zk.
786       try {
787         assignmentManager.getZKTable().setEnabledTable(tableName);
788       } catch (KeeperException e) {
789         throw new IOException("Unable to ensure that the table will be" +
790             " enabled because of a ZooKeeper issue", e);
791       }
792 
793       // 2. Create HRegion
794       HRegion region = HRegion.createHRegion(newRegion,
795           fileSystemManager.getRootDir(), conf);
796 
797       // 3. Insert into META
798       MetaEditor.addRegionToMeta(catalogTracker, region.getRegionInfo());
799 
800       // 4. Close the new region to flush to disk.  Close log file too.
801       region.close();
802       region.getLog().closeAndDelete();
803 
804       // 5. Trigger immediate assignment of this region
805       assignmentManager.assign(region.getRegionInfo(), true);
806     }
807 
808     // 5. If sync, wait for assignment of regions
809     if(sync) {
810       LOG.debug("Waiting for " + newRegions.length + " region(s) to be " +
811           "assigned before returning");
812       for(HRegionInfo regionInfo : newRegions) {
813         try {
814           assignmentManager.waitForAssignment(regionInfo);
815         } catch (InterruptedException e) {
816           LOG.info("Interrupted waiting for region to be assigned during " +
817               "create table call");
818           return;
819         }
820       }
821     }
822   }

858   /**
859    * Return the region and current deployment for the region containing
860    * the given row. If the region cannot be found, returns null. If it
861    * is found, but not currently deployed, the second element of the pair
862    * may be null.
863    */
864   Pair<HRegionInfo,HServerAddress> getTableRegionForRow(
865       final byte [] tableName, final byte [] rowKey)
866   throws IOException {
867     final AtomicReference<Pair<HRegionInfo, HServerAddress>> result =
868       new AtomicReference<Pair<HRegionInfo, HServerAddress>>(null);
869 
870     MetaScannerVisitor visitor =
871       new MetaScannerVisitor() {
872         @Override
873         public boolean processRow(Result data) throws IOException {
874           if (data == null || data.size() <= 0) {
875             return true;
876           }
877           Pair<HRegionInfo, HServerAddress> pair =
878             MetaReader.metaRowToRegionPair(data);
879           if (pair == null) {
880             return false;
881           }
882           if (!Bytes.equals(pair.getFirst().getTableDesc().getName(),
883                 tableName)) {
884             return false;
885           }
886           result.set(pair);
887           return true;
888         }
889     };
890 
891     MetaScanner.metaScan(conf, visitor, tableName, rowKey, 1);
892     return result.get();
893   }
