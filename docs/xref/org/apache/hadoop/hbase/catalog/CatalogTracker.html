43  /**
44   * Tracks the availability of the catalog tables <code>-ROOT-</code> and
45   * <code>.META.</code>.
46   * 
47   * This class is "read-only" in that the locations of the catalog tables cannot
48   * be explicitly set.  Instead, ZooKeeper is used to learn of the availability
49   * and location of <code>-ROOT-</code>.  <code>-ROOT-</code> is used to learn of
50   * the location of <code>.META.</code>  If not available in <code>-ROOT-</code>,
51   * ZooKeeper is used to monitor for a new location of <code>.META.</code>.
52   *
53   * <p>Call {@link #start()} to start up operation.  Call {@link #stop()}} to
54   * interrupt waits and close up shop.
55   */
56  public class CatalogTracker {
57    private static final Log LOG = LogFactory.getLog(CatalogTracker.class);
58    private final HConnection connection;
59    private final ZooKeeperWatcher zookeeper;
60    private final RootRegionTracker rootRegionTracker;
61    private final MetaNodeTracker metaNodeTracker;
62    private final AtomicBoolean metaAvailable = new AtomicBoolean(false);
63    /**
64     * Do not clear this address once set.  Let it be cleared by
65     * {@link #setMetaLocation(HServerAddress)} only.  Its needed when we do
66     * server shutdown processing -- we need to know who had .META. last.  If you
67     * want to know if the address is good, rely on {@link #metaAvailable} value.
68     */
69    private HServerAddress metaLocation;
70    private final int defaultTimeout;
71    private boolean stopped = false;

138   /**

143     LOG.debug("Stopping catalog tracker " + this);
144     this.stopped = true;
145     this.rootRegionTracker.stop();
146     this.metaNodeTracker.stop();
147     // Call this and it will interrupt any ongoing waits on meta.

246   /**
247    * Gets a connection to the server currently hosting <code>.META.</code> or
248    * null if location is not currently available.
249    * <p>
250    * If a location is known, a connection to the cached location is returned.
251    * If refresh is true, the cached connection is verified first before
252    * returning.  If the connection is not valid, it is reset and rechecked.
253    * <p>
254    * If no location for meta is currently known, method checks ROOT for a new
255    * location, verifies META is currently there, and returns a cached connection
256    * to the server hosting META.
257    *
258    * @return connection to server hosting meta, null if location not available
259    * @throws IOException
260    * @throws InterruptedException 
261    */
262   private HRegionInterface getMetaServerConnection(boolean refresh)
263   throws IOException, InterruptedException {
264     synchronized (metaAvailable) {
265       if (metaAvailable.get()) {
266         HRegionInterface current = getCachedConnection(metaLocation);
267         if (!refresh) {
268           return current;
269         }
270         if (verifyRegionLocation(current, this.metaLocation, META_REGION)) {
271           return current;
272         }
273         resetMetaLocation();
274       }
275       HRegionInterface rootConnection = getRootServerConnection();
276       if (rootConnection == null) {
277         return null;
278       }
279       HServerAddress newLocation = MetaReader.readMetaLocation(rootConnection);
280       if (newLocation == null) {
281         return null;
282       }
283       HRegionInterface newConnection = getCachedConnection(newLocation);
284       if (verifyRegionLocation(newConnection, this.metaLocation, META_REGION)) {
285         setMetaLocation(newLocation);
286         return newConnection;
287       }
288       return null;
289     }
290   }

305   /**
306    * Gets the current location for <code>.META.</code> if available and waits
307    * for up to the specified timeout if not immediately available.  Throws an
308    * exception if timed out waiting.  This method differs from {@link #waitForMeta()}
309    * in that it will go ahead and verify the location gotten from ZooKeeper by
310    * trying to use returned connection.
311    * @param timeout maximum time to wait for meta availability, in milliseconds
312    * @return location of meta
313    * @throws InterruptedException if interrupted while waiting
314    * @throws IOException unexpected exception connecting to meta server
315    * @throws NotAllMetaRegionsOnlineException if meta not available before
316    *                                          timeout
317    */
318   public HServerAddress waitForMeta(long timeout)
319   throws InterruptedException, IOException, NotAllMetaRegionsOnlineException {
320     long stop = System.currentTimeMillis() + timeout;
321     synchronized (metaAvailable) {
322       if (getMetaServerConnection(true) != null) {
323         return metaLocation;
324       }
325       while(!stopped && !metaAvailable.get() &&
326           (timeout == 0 || System.currentTimeMillis() < stop)) {
327         metaAvailable.wait(timeout);
328       }
329       if (getMetaServerConnection(true) == null) {
330         throw new NotAllMetaRegionsOnlineException(
331             "Timed out (" + timeout + "ms)");
332       }
333       return metaLocation;
334     }
335   }

380   private HRegionInterface getCachedConnection(HServerAddress address)
381   throws IOException {
382     HRegionInterface protocol = null;
383     try {
384       protocol = connection.getHRegionConnection(address, false);
385     } catch (RetriesExhaustedException e) {
386       if (e.getCause() != null && e.getCause() instanceof ConnectException) {
387         // Catch this; presume it means the cached connection has gone bad.
388       } else {
389         throw e;
390       }
391     } catch (IOException ioe) {
392       Throwable cause = ioe.getCause();
393       if (cause != null && cause instanceof EOFException) {
394         // Catch. Other end disconnected us.
395       } else if (cause != null && cause.getMessage() != null &&
396         cause.getMessage().toLowerCase().contains("connection reset")) {
397         // Catch. Connection reset.
398       } else {
399         throw ioe;
400       }
401       
402     }
403     return protocol;
404   }

406   private boolean verifyRegionLocation(HRegionInterface metaServer,
407       final HServerAddress address,
408       byte [] regionName)
409   throws IOException {
410     if (metaServer == null) {
411       LOG.info("Passed metaserver is null");
412       return false;
413     }
414     Throwable t = null;
415     try {
416       return metaServer.getRegionInfo(regionName) != null;
417     } catch (ConnectException e) {
418       t = e;
419     } catch (RemoteException e) {
420       IOException ioe = e.unwrapRemoteException();
421       if (ioe instanceof NotServingRegionException) {
422         t = ioe;
423       } else {
424         throw e;
425       }
426     } catch (IOException e) {
427       Throwable cause = e.getCause();
428       if (cause != null && cause instanceof EOFException) {
429         t = cause;
430       } else if (cause != null && cause.getMessage() != null
431           && cause.getMessage().contains("Connection reset")) {
432         t = cause;
433       } else {
434         throw e;
435       }
436     }
437     LOG.info("Failed verification of " + Bytes.toString(regionName) +
438       " at address=" + address + "; " + t);
439     return false;
440   }

442   /**
443    * Verify <code>-ROOT-</code> is deployed and accessible.
444    * @param timeout How long to wait on zk for root address (passed through to
445    * the internal call to {@link #waitForRootServerConnection(long)}.
446    * @return True if the <code>-ROOT-</code> location is healthy.
447    * @throws IOException
448    * @throws InterruptedException 
449    */
450   public boolean verifyRootRegionLocation(final long timeout)
451   throws InterruptedException, IOException {
452     HRegionInterface connection = null;
453     try {
454       connection = waitForRootServerConnection(timeout);
455     } catch (NotAllMetaRegionsOnlineException e) {
456       // Pass
457     } catch (org.apache.hadoop.hbase.ipc.ServerNotRunningException e) {
458       // Pass -- remote server is not up so can't be carrying root
459     } catch (IOException e) {
460       // Unexpected exception
461       throw e;
462     }
463     return (connection == null)? false:
464       verifyRegionLocation(connection,this.rootRegionTracker.getRootRegionLocation(),
465         HRegionInfo.ROOT_REGIONINFO.getRegionName());
466   }
