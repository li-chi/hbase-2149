25  import org.apache.commons.logging.LogFactory;
26  import org.apache.hadoop.conf.Configuration;

28  /**
29   * Adds HBase configuration files to a Configuration
30   */
31  public class HBaseConfiguration extends Configuration {
32  
33    private static final Log LOG = LogFactory.getLog(HBaseConfiguration.class);
34  
35    /**
36     * Instantinating HBaseConfiguration() is deprecated. Please use
37     * HBaseConfiguration#create() to construct a plain Configuration
38     */
39    @Deprecated
40    public HBaseConfiguration() {
41      //TODO:replace with private constructor, HBaseConfiguration should not extend Configuration
42      super();
43      addHbaseResources(this);
44      LOG.warn("instantiating HBaseConfiguration() is deprecated. Please use" +
45      		" HBaseConfiguration#create() to construct a plain Configuration");
46    }
47  
48    /**
49     * Instantiating HBaseConfiguration() is deprecated. Please use
50     * HBaseConfiguration#create(conf) to construct a plain Configuration
51     */
52    @Deprecated
53    public HBaseConfiguration(final Configuration c) {
54      //TODO:replace with private constructor
55      this();
56      for (Entry<String, String>e: c) {
57        set(e.getKey(), e.getValue());
58      }
59    }
60  
61    public static Configuration addHbaseResources(Configuration conf) {
62      conf.addResource("hbase-default.xml");
63      conf.addResource("hbase-site.xml");
64      return conf;
65    }
66  
67    /**
68     * Creates a Configuration with HBase resources
69     * @return a Configuration with HBase resources
70     */
71    public static Configuration create() {
72      Configuration conf = new Configuration();
73      return addHbaseResources(conf);
74    }
75  
76    /**
77     * Creates a clone of passed configuration.
78     * @param that Configuration to clone.
79     * @return a Configuration created with the hbase-*.xml files plus
80     * the given configuration.
81     */
82    public static Configuration create(final Configuration that) {
83      Configuration conf = create();
84      for (Entry<String, String>e: that) {
85        conf.set(e.getKey(), e.getValue());
86      }
87      return conf;
88    }
89  }
