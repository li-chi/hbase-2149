52  /**
53   * Executes region split as a "transaction".  Call {@link #prepare()} to setup
54   * the transaction, {@link #execute(OnlineRegions)} to run the transaction and
55   * {@link #rollback(OnlineRegions)} to cleanup if execute fails.
56   *
57   * <p>Here is an example of how you would use this class:
58   * <pre>
59   *  SplitTransaction st = new SplitTransaction(this.conf, parent, midKey)
60   *  if (!st.prepare()) return;
61   *  try {
62   *    st.execute(myOnlineRegions);
63   *  } catch (IOException ioe) {
64   *    try {
65   *      st.rollback(myOnlineRegions);
66   *      return;
67   *    } catch (RuntimeException e) {
68   *      myAbortable.abort("Failed split, abort");
69   *    }
70   *  }
71   * </Pre>
72   * <p>This class is not thread safe.  Caller needs ensure split is run by
73   * one thread only.
74   */
75  class SplitTransaction {

138   /**
139    * Does checks on split inputs.
140    * @return <code>true</code> if the region is splittable else
141    * <code>false</code> if it is not (e.g. its already closed, etc.).
142    */
143   public boolean prepare() {
144     if (this.parent.isClosed() || this.parent.isClosing()) return false;
145     HRegionInfo hri = this.parent.getRegionInfo();
146     parent.prepareToSplit();
147     // Check splitrow.
148     byte [] startKey = hri.getStartKey();
149     byte [] endKey = hri.getEndKey();
150     if (Bytes.equals(startKey, splitrow) ||
151         !this.parent.getRegionInfo().containsRow(splitrow)) {
152       LOG.info("Split row is not inside region key range or is equal to " +
153           "startkey: " + Bytes.toString(this.splitrow));
154       return false;
155     }
156     long rid = getDaughterRegionIdTimestamp(hri);
157     this.hri_a = new HRegionInfo(hri.getTableDesc(), startKey, this.splitrow,
158       false, rid);
159     this.hri_b = new HRegionInfo(hri.getTableDesc(), this.splitrow, endKey,
160       false, rid);
161     return true;
162   }

164   /**
165    * Calculate daughter regionid to use.
166    * @param hri Parent {@link HRegionInfo}
167    * @return Daughter region id (timestamp) to use.
168    */
169   private static long getDaughterRegionIdTimestamp(final HRegionInfo hri) {
170     long rid = EnvironmentEdgeManager.currentTimeMillis();
171     // Regionid is timestamp.  Can't be less than that of parent else will insert
172     // at wrong location in .META. (See HBASE-710).
173     if (rid < hri.getRegionId()) {
174       LOG.warn("Clock skew; parent regions id is " + hri.getRegionId() +
175         " but current time here is " + rid);
176       rid = hri.getRegionId() + 1;
177     }
178     return rid;
179   }

181   /**
182    * Run the transaction.
183    * @param server Hosting server instance.
184    * @param services Used to online/offline regions.
185    * @throws IOException If thrown, transaction failed. Call {@link #rollback(OnlineRegions)}
186    * @return Regions created
187    * @see #rollback(OnlineRegions)
188    */
189   PairOfSameType<HRegion> execute(final Server server,
190       final RegionServerServices services)
191   throws IOException {
192     LOG.info("Starting split of region " + this.parent);
193     if ((server != null && server.isStopped()) ||
194         (services != null && services.isStopping())) {
195       throw new IOException("Server is stopped or stopping");
196     }
197     assert !this.parent.lock.writeLock().isHeldByCurrentThread() : "Unsafe to hold write lock while performing RPCs";
198 
199     // If true, no cluster to write meta edits into.
200     boolean testing = server == null? true:
201       server.getConfiguration().getBoolean("hbase.testing.nocluster", false);
202     this.fileSplitTimeout = testing ? this.fileSplitTimeout :
203         server.getConfiguration().getLong(
204             "hbase.regionserver.fileSplitTimeout", this.fileSplitTimeout);
205 
206     createSplitDir(this.parent.getFilesystem(), this.splitdir);
207     this.journal.add(JournalEntry.CREATE_SPLIT_DIR);
208 
209     List<StoreFile> hstoreFilesToSplit = this.parent.close(false);
210     if (hstoreFilesToSplit == null) {
211       // The region was closed by a concurrent thread.  We can't continue
212       // with the split, instead we must just abandon the split.  If we
213       // reopen or split this could cause problems because the region has
214       // probably already been moved to a different server, or is in the
215       // process of moving to a different server.
216       throw new IOException("Failed to close region: already closed by " +
217         "another thread");
218     }
219     this.journal.add(JournalEntry.CLOSED_PARENT_REGION);
220 
221     if (!testing) {
222       services.removeFromOnlineRegions(this.parent.getRegionInfo().getEncodedName());
223     }
224     this.journal.add(JournalEntry.OFFLINED_PARENT);
225     
226     // TODO: If the below were multithreaded would we complete steps in less
227     // elapsed time?  St.Ack 20100920
228 
229     splitStoreFiles(this.splitdir, hstoreFilesToSplit);
230     // splitStoreFiles creates daughter region dirs under the parent splits dir
231     // Nothing to unroll here if failure -- clean up of CREATE_SPLIT_DIR will
232     // clean this up.
233 
234     // Log to the journal that we are creating region A, the first daughter
235     // region.  We could fail halfway through.  If we do, we could have left
236     // stuff in fs that needs cleanup -- a storefile or two.  Thats why we
237     // add entry to journal BEFORE rather than AFTER the change.
238     this.journal.add(JournalEntry.STARTED_REGION_A_CREATION);
239     HRegion a = createDaughterRegion(this.hri_a, this.parent.flushRequester);
240 
241     // Ditto
242     this.journal.add(JournalEntry.STARTED_REGION_B_CREATION);
243     HRegion b = createDaughterRegion(this.hri_b, this.parent.flushRequester);
244 
245     // Edit parent in meta
246     if (!testing) {
247       MetaEditor.offlineParentInMeta(server.getCatalogTracker(),
248         this.parent.getRegionInfo(), a.getRegionInfo(), b.getRegionInfo());
249     }
250 
251     // The is the point of no return.  We are committed to the split now.  We
252     // have still the daughter regions to open but meta has been changed.
253     // If we fail from here on out, we can not rollback so, we'll just abort.
254     // The meta has been changed though so there will need to be a fixup run
255     // during processing of the crashed server by master (TODO: Verify this in place).
256 
257     // TODO: Could we be smarter about the sequence in which we do these steps?
258 
259     if (!testing) {
260       // Open daughters in parallel.
261       DaughterOpener aOpener = new DaughterOpener(server, services, a);
262       DaughterOpener bOpener = new DaughterOpener(server, services, b);
263       aOpener.start();
264       bOpener.start();
265       try {
266         aOpener.join();
267         bOpener.join();
268       } catch (InterruptedException e) {
269         server.abort("Exception running daughter opens", e);
270       }
271     }
272 
273     // Leaving here, the splitdir with its dross will be in place but since the
274     // split was successful, just leave it; it'll be cleaned when parent is
275     // deleted and cleaned up.
276     return new PairOfSameType<HRegion>(a, b);
277   }

279   class DaughterOpener extends Thread {
280     private final RegionServerServices services;
281     private final Server server;
282     private final HRegion r;
283 
284     DaughterOpener(final Server s, final RegionServerServices services,
285         final HRegion r) {
286       super(s.getServerName() + "-daughterOpener=" + r.getRegionInfo().getEncodedName());
287       setDaemon(true);
288       this.services = services;
289       this.server = s;
290       this.r = r;
291     }
292 
293     @Override
294     public void run() {
295       try {
296         openDaughterRegion(this.server, this.services, r);
297       } catch (Throwable t) {
298         this.server.abort("Failed open of daughter " +
299           this.r.getRegionInfo().getRegionNameAsString(), t);
300       }
301     }
302   }

304   /**
305    * Open daughter regions, add them to online list and update meta.
306    * @param server
307    * @param services
308    * @param daughter
309    * @throws IOException
310    * @throws KeeperException
311    */
312   void openDaughterRegion(final Server server,
313       final RegionServerServices services, final HRegion daughter)
314   throws IOException, KeeperException {
315     if (server.isStopped() || services.isStopping()) {
316       MetaEditor.addDaughter(server.getCatalogTracker(),
317         daughter.getRegionInfo(), null);
318       LOG.info("Not opening daughter " +
319         daughter.getRegionInfo().getRegionNameAsString() +
320         " because stopping=" + services.isStopping() + ", stopped=" +
321         server.isStopped());
322       return;
323     }
324     HRegionInfo hri = daughter.getRegionInfo();
325     LoggingProgressable reporter =
326       new LoggingProgressable(hri, server.getConfiguration());
327     HRegion r = daughter.openHRegion(reporter);
328     services.postOpenDeployTasks(r, server.getCatalogTracker(), true);
329   }

392   private void splitStoreFiles(final Path splitdir,
393     final List<StoreFile> hstoreFilesToSplit)
394   throws IOException {
395     if (hstoreFilesToSplit == null) {
396       // Could be null because close didn't succeed -- for now consider it fatal
397       throw new IOException("Close returned empty list of StoreFiles");
398     }
399     // The following code sets up a thread pool executor with as many slots as
400     // there's files to split. It then fires up everything, waits for
401     // completion and finally checks for any exception
402     int nbFiles = hstoreFilesToSplit.size();
403     ThreadFactoryBuilder builder = new ThreadFactoryBuilder();
404     builder.setNameFormat("StoreFileSplitter-%1$d");
405     ThreadFactory factory = builder.build();
406     ThreadPoolExecutor threadPool =
407       (ThreadPoolExecutor) Executors.newFixedThreadPool(nbFiles, factory);
408     List<Future<Void>> futures = new ArrayList<Future<Void>>(nbFiles);
409 
410      // Split each store file.
411     for (StoreFile sf: hstoreFilesToSplit) {
412       //splitStoreFile(sf, splitdir);
413       StoreFileSplitter sfs = new StoreFileSplitter(sf, splitdir);
414       futures.add(threadPool.submit(sfs));
415     }
416     // Shutdown the pool
417     threadPool.shutdown();
418 
419     // Wait for all the tasks to finish
420     try {
421       boolean stillRunning = !threadPool.awaitTermination(
422           this.fileSplitTimeout, TimeUnit.MILLISECONDS);
423       if (stillRunning) {
424         threadPool.shutdownNow();
425         throw new IOException("Took too long to split the" +
426             " files and create the references, aborting split");
427       }
428     } catch (InterruptedException e) {
429       Thread.currentThread().interrupt();
430       throw new IOException("Interrupted while waiting for file splitters", e);
431     }
432 
433     // Look for any exception
434     for (Future future : futures) {
435       try {
436         future.get();
437       } catch (InterruptedException e) {
438         Thread.currentThread().interrupt();
439         throw new IOException(
440             "Interrupted while trying to get the results of file splitters", e);
441       } catch (ExecutionException e) {
442         throw new IOException(e);
443       }
444     }
445   }

447   private void splitStoreFile(final StoreFile sf, final Path splitdir)
448   throws IOException {
449     FileSystem fs = this.parent.getFilesystem();
450     byte [] family = sf.getFamily();
451     String encoded = this.hri_a.getEncodedName();
452     Path storedir = Store.getStoreHomedir(splitdir, encoded, family);
453     StoreFile.split(fs, storedir, sf, this.splitrow, Range.bottom);
454     encoded = this.hri_b.getEncodedName();
455     storedir = Store.getStoreHomedir(splitdir, encoded, family);
456     StoreFile.split(fs, storedir, sf, this.splitrow, Range.top);
457   }

528   /**
529    * @param or Object that can online/offline parent region.  Can be passed null
530    * by unit tests.
531    * @return The region we were splitting
532    * @throws IOException If thrown, rollback failed.  Take drastic action.
533    */
534   public void rollback(final OnlineRegions or) throws IOException {
535     FileSystem fs = this.parent.getFilesystem();
536     ListIterator<JournalEntry> iterator =
537       this.journal.listIterator(this.journal.size());
538     while (iterator.hasPrevious()) {
539       JournalEntry je = iterator.previous();
540       switch(je) {
541       case CREATE_SPLIT_DIR:
542         cleanupSplitDir(fs, this.splitdir);
543         break;
544 
545       case CLOSED_PARENT_REGION:
546         // So, this returns a seqid but if we just closed and then reopened, we
547         // should be ok. On close, we flushed using sequenceid obtained from
548         // hosting regionserver so no need to propagate the sequenceid returned
549         // out of initialize below up into regionserver as we normally do.
550         // TODO: Verify.
551         this.parent.initialize();
552         break;
553 
554       case STARTED_REGION_A_CREATION:
555         cleanupDaughterRegion(fs, this.parent.getTableDir(),
556           this.hri_a.getEncodedName());
557         break;
558 
559       case STARTED_REGION_B_CREATION:
560         cleanupDaughterRegion(fs, this.parent.getTableDir(),
561           this.hri_b.getEncodedName());
562         break;
563 
564       case OFFLINED_PARENT:
565         if (or != null) or.addToOnlineRegions(this.parent);
566         break;
567 
568       default:
569         throw new RuntimeException("Unhandled journal entry: " + je);
570       }
571     }
572   }

587   /**
588    * Clean up any split detritus that may have been left around from previous
589    * split attempts.
590    * Call this method on initial region deploy.  Cleans up any mess
591    * left by previous deploys of passed <code>r</code> region.
592    * @param r
593    * @throws IOException 
594    */
595   static void cleanupAnySplitDetritus(final HRegion r) throws IOException {
596     Path splitdir = getSplitDir(r);
597     FileSystem fs = r.getFilesystem();
598     if (!fs.exists(splitdir)) return;
599     // Look at the splitdir.  It could have the encoded names of the daughter
600     // regions we tried to make.  See if the daughter regions actually got made
601     // out under the tabledir.  If here under splitdir still, then the split did
602     // not complete.  Try and do cleanup.  This code WILL NOT catch the case
603     // where we successfully created daughter a but regionserver crashed during
604     // the creation of region b.  In this case, there'll be an orphan daughter
605     // dir in the filesystem.  TOOD: Fix.
606     FileStatus [] daughters = fs.listStatus(splitdir, new FSUtils.DirFilter(fs));
607     for (int i = 0; i < daughters.length; i++) {
608       cleanupDaughterRegion(fs, r.getTableDir(),
609         daughters[i].getPath().getName());
610     }
611     cleanupSplitDir(r.getFilesystem(), splitdir);
612     LOG.info("Cleaned up old failed split transaction detritus: " + splitdir);
613   }
