154    * @throws java.lang.Exception
155    */
156   @Before
157   public void setUp() throws Exception {
158 
159     // Starting and stopping replication can make us miss new logs,
160     // rolling like this makes sure the most recent one gets added to the queue
161     for ( JVMClusterUtil.RegionServerThread r :
162         utility1.getHBaseCluster().getRegionServerThreads()) {
163       r.getRegionServer().getWAL().rollWriter();
164     }
165     utility1.truncateTable(tableName);
166     // truncating the table will send one Delete per row to the slave cluster
167     // in an async fashion, which is why we cannot just call truncateTable on
168     // utility2 since late writes could make it to the slave in some way.
169     // Instead, we truncate the first table and wait for all the Deletes to
170     // make it to the slave.
171     Scan scan = new Scan();
172     int lastCount = 0;
173     for (int i = 0; i < NB_RETRIES; i++) {
174       if (i==NB_RETRIES-1) {
175         fail("Waited too much time for truncate");
176       }
177       ResultScanner scanner = htable2.getScanner(scan);
178       Result[] res = scanner.next(NB_ROWS_IN_BIG_BATCH);
179       scanner.close();
180       if (res.length != 0) {
181        if (res.length < lastCount) {
182           i--; // Don't increment timeout if we make progress
183         }
184         lastCount = res.length;
185         LOG.info("Still got " + res.length + " rows");
186         Thread.sleep(SLEEP_TIME);
187       } else {
188         break;
189       }
190     }
191   }

517   /**
518    * Load up multiple tables over 2 region servers and kill a source during
519    * the upload. The failover happens internally.
520    * @throws Exception
521    */
522   @Test
523   public void queueFailover() throws Exception {
524     utility1.createMultiRegions(htable1, famName);
525 
526     // killing the RS with .META. can result into failed puts until we solve
527     // IO fencing
528     int rsToKill1 =
529         utility1.getHBaseCluster().getServerWithMeta() == 0 ? 1 : 0;
530     int rsToKill2 =
531         utility2.getHBaseCluster().getServerWithMeta() == 0 ? 1 : 0;
532 
533     // Takes about 20 secs to run the full loading, kill around the middle
534     Thread killer1 = killARegionServer(utility1, 7500, rsToKill1);
535     Thread killer2 = killARegionServer(utility2, 10000, rsToKill2);
536 
537     LOG.info("Start loading table");
538     int initialCount = utility1.loadTable(htable1, famName);
539     LOG.info("Done loading table");
540     killer1.join(5000);
541     killer2.join(5000);
542     LOG.info("Done waiting for threads");
543 
544     Result[] res;
545     while (true) {
546       try {
547         Scan scan = new Scan();
548         ResultScanner scanner = htable1.getScanner(scan);
549         res = scanner.next(initialCount);
550         scanner.close();
551         break;
552       } catch (UnknownScannerException ex) {
553         LOG.info("Cluster wasn't ready yet, restarting scanner");
554       }
555     }
556     // Test we actually have all the rows, we may miss some because we
557     // don't have IO fencing.
558     if (res.length != initialCount) {
559       LOG.warn("We lost some rows on the master cluster!");
560       // We don't really expect the other cluster to have more rows
561       initialCount = res.length;
562     }
563 
564     Scan scan2 = new Scan();
565 
566     int lastCount = 0;
567 
568     for (int i = 0; i < NB_RETRIES; i++) {
569       if (i==NB_RETRIES-1) {
570         fail("Waited too much time for queueFailover replication");
571       }
572       ResultScanner scanner2 = htable2.getScanner(scan2);
573       Result[] res2 = scanner2.next(initialCount * 2);
574       scanner2.close();
575       if (res2.length < initialCount) {
576         if (lastCount < res2.length) {
577           i--; // Don't increment timeout if we make progress
578         }
579         lastCount = res2.length;
580         LOG.info("Only got " + lastCount + " rows instead of " +
581             initialCount + " current i=" + i);
582         Thread.sleep(SLEEP_TIME*2);
583       } else {
584         break;
585       }
586     }
587   }

589   private static Thread killARegionServer(final HBaseTestingUtility utility,
590                                    final long timeout, final int rs) {
591     Thread killer = new Thread() {
592       public void run() {
593         try {
594           Thread.sleep(timeout);
595           utility.expireRegionServerSession(rs);
596         } catch (Exception e) {
597           LOG.error(e);
598         }
599       }
600     };
601     killer.start();
602     return killer;
603   }
