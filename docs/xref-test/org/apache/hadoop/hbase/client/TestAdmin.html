152   @Test public void testChangeTableSchema() throws IOException {
153     final byte [] tableName = Bytes.toBytes("changeTableSchema");
154     HTableDescriptor [] tables = admin.listTables();
155     int numTables = tables.length;
156     TEST_UTIL.createTable(tableName, HConstants.CATALOG_FAMILY);
157     tables = this.admin.listTables();
158     assertEquals(numTables + 1, tables.length);
159 
160     // FIRST, do htabledescriptor changes.
161     HTableDescriptor htd = this.admin.getTableDescriptor(tableName);
162     // Make a copy and assert copy is good.
163     HTableDescriptor copy = new HTableDescriptor(htd);
164     assertTrue(htd.equals(copy));
165     // Now amend the copy. Introduce differences.
166     long newFlushSize = htd.getMemStoreFlushSize() / 2;
167     copy.setMemStoreFlushSize(newFlushSize);
168     final String key = "anyoldkey";
169     assertTrue(htd.getValue(key) == null);
170     copy.setValue(key, key);
171     boolean expectedException = false;
172     try {
173       this.admin.modifyTable(tableName, copy);
174     } catch (TableNotDisabledException re) {
175       expectedException = true;
176     }
177     assertTrue(expectedException);
178     this.admin.disableTable(tableName);
179     assertTrue(this.admin.isTableDisabled(tableName));
180     modifyTable(tableName, copy);
181     HTableDescriptor modifiedHtd = this.admin.getTableDescriptor(tableName);
182     // Assert returned modifiedhcd is same as the copy.
183     assertFalse(htd.equals(modifiedHtd));
184     assertTrue(copy.equals(modifiedHtd));
185     assertEquals(newFlushSize, modifiedHtd.getMemStoreFlushSize());
186     assertEquals(key, modifiedHtd.getValue(key));
187 
188     // Reenable table to test it fails if not disabled.
189     this.admin.enableTable(tableName);
190     assertFalse(this.admin.isTableDisabled(tableName));
191 
192     // Now work on column family changes.
193     int countOfFamilies = modifiedHtd.getFamilies().size();
194     assertTrue(countOfFamilies > 0);
195     HColumnDescriptor hcd = modifiedHtd.getFamilies().iterator().next();
196     int maxversions = hcd.getMaxVersions();
197     final int newMaxVersions = maxversions + 1;
198     hcd.setMaxVersions(newMaxVersions);
199     final byte [] hcdName = hcd.getName();
200     expectedException = false;
201     try {
202       this.admin.modifyColumn(tableName, hcd);
203     } catch (TableNotDisabledException re) {
204       expectedException = true;
205     }
206     assertTrue(expectedException);
207     this.admin.disableTable(tableName);
208     assertTrue(this.admin.isTableDisabled(tableName));
209     // Modify Column is synchronous
210     this.admin.modifyColumn(tableName, hcd);
211     modifiedHtd = this.admin.getTableDescriptor(tableName);
212     HColumnDescriptor modifiedHcd = modifiedHtd.getFamily(hcdName);
213     assertEquals(newMaxVersions, modifiedHcd.getMaxVersions());
214 
215     // Try adding a column
216     // Reenable table to test it fails if not disabled.
217     this.admin.enableTable(tableName);
218     assertFalse(this.admin.isTableDisabled(tableName));
219     final String xtracolName = "xtracol";
220     HColumnDescriptor xtracol = new HColumnDescriptor(xtracolName);
221     xtracol.setValue(xtracolName, xtracolName);
222     try {
223       this.admin.addColumn(tableName, xtracol);
224     } catch (TableNotDisabledException re) {
225       expectedException = true;
226     }
227     assertTrue(expectedException);
228     this.admin.disableTable(tableName);
229     assertTrue(this.admin.isTableDisabled(tableName));
230     this.admin.addColumn(tableName, xtracol);
231     modifiedHtd = this.admin.getTableDescriptor(tableName);
232     hcd = modifiedHtd.getFamily(xtracol.getName());
233     assertTrue(hcd != null);
234     assertTrue(hcd.getValue(xtracolName).equals(xtracolName));
235 
236     // Delete the just-added column.
237     this.admin.deleteColumn(tableName, xtracol.getName());
238     modifiedHtd = this.admin.getTableDescriptor(tableName);
239     hcd = modifiedHtd.getFamily(xtracol.getName());
240     assertTrue(hcd == null);
241 
242     // Delete the table
243     this.admin.deleteTable(tableName);
244     this.admin.listTables();
245     assertFalse(this.admin.tableExists(tableName));
246   }

248   /**
249    * Modify table is async so wait on completion of the table operation in master.
250    * @param tableName
251    * @param htd
252    * @throws IOException
253    */
254   private void modifyTable(final byte [] tableName, final HTableDescriptor htd)
255   throws IOException {
256     MasterServices services = TEST_UTIL.getMiniHBaseCluster().getMaster();
257     ExecutorService executor = services.getExecutorService();
258     AtomicBoolean done = new AtomicBoolean(false);
259     executor.registerListener(EventType.C_M_MODIFY_TABLE, new DoneListener(done));
260     this.admin.modifyTable(tableName, htd);
261     while (!done.get()) {
262       synchronized (done) {
263         try {
264           done.wait(1000);
265         } catch (InterruptedException e) {
266           e.printStackTrace();
267         }
268       }
269     }
270     executor.unregisterListener(EventType.C_M_MODIFY_TABLE);
271   }

273   /**
274    * Listens for when an event is done in Master.
275    */
276   static class DoneListener implements EventHandler.EventHandlerListener {
277     private final AtomicBoolean done;
278 
279     DoneListener(final AtomicBoolean done) {
280       super();
281       this.done = done;
282     }
283 
284     @Override
285     public void afterProcess(EventHandler event) {
286       this.done.set(true);
287       synchronized (this.done) {
288         // Wake anyone waiting on this value to change.
289         this.done.notifyAll();
290       }
291     }
292 
293     @Override
294     public void beforeProcess(EventHandler event) {
295       // continue
296     }
297   }

470   /**
471    * Tests forcing split from client and having scanners successfully ride over split.
472    * @throws Exception
473    * @throws IOException
474    */
475   @Test
476   public void testForceSplit() throws Exception {
477     byte [] familyName = HConstants.CATALOG_FAMILY;
478     byte [] tableName = Bytes.toBytes("testForceSplit");
479     final HTable table = TEST_UTIL.createTable(tableName, familyName);
480     byte[] k = new byte[3];
481     int rowCount = 0;
482     for (byte b1 = 'a'; b1 < 'z'; b1++) {
483       for (byte b2 = 'a'; b2 < 'z'; b2++) {
484         for (byte b3 = 'a'; b3 < 'z'; b3++) {
485           k[0] = b1;
486           k[1] = b2;
487           k[2] = b3;
488           Put put = new Put(k);
489           put.add(familyName, new byte[0], k);
490           table.put(put);
491           rowCount++;
492         }
493       }
494     }
495 
496     // get the initial layout (should just be one region)
497     Map<HRegionInfo,HServerAddress> m = table.getRegionsInfo();
498     System.out.println("Initial regions (" + m.size() + "): " + m);
499     assertTrue(m.size() == 1);
500 
501     // Verify row count
502     Scan scan = new Scan();
503     ResultScanner scanner = table.getScanner(scan);
504     int rows = 0;
505     for(@SuppressWarnings("unused") Result result : scanner) {
506       rows++;
507     }
508     scanner.close();
509     assertEquals(rowCount, rows);
510 
511     // Have an outstanding scan going on to make sure we can scan over splits.
512     scan = new Scan();
513     scanner = table.getScanner(scan);
514     // Scan first row so we are into first region before split happens.
515     scanner.next();
516 
517     final AtomicInteger count = new AtomicInteger(0);
518     Thread t = new Thread("CheckForSplit") {
519       public void run() {
520         for (int i = 0; i < 20; i++) {
521           try {
522             sleep(1000);
523           } catch (InterruptedException e) {
524             continue;
525           }
526           // check again    table = new HTable(conf, tableName);
527           Map<HRegionInfo, HServerAddress> regions = null;
528           try {
529             regions = table.getRegionsInfo();
530           } catch (IOException e) {
531             e.printStackTrace();
532           }
533           if (regions == null) continue;
534           count.set(regions.size());
535           if (count.get() >= 2) break;
536           LOG.debug("Cycle waiting on split");
537         }
538       }
539     };
540     t.start();
541     // Split the table
542     this.admin.split(Bytes.toString(tableName));
543     t.join();
544 
545     // Verify row count
546     rows = 1; // We counted one row above.
547     for (@SuppressWarnings("unused") Result result : scanner) {
548       rows++;
549       if (rows > rowCount) {
550         scanner.close();
551         assertTrue("Scanned more than expected (" + rowCount + ")", false);
552       }
553     }
554     scanner.close();
555     assertEquals(rowCount, rows);
556   }
