29   * Testing Testing {@link Pair#equals(Object)} for deep checking of arrays

31  public class TestPairEquals {
32  
33    /**
34     * Testing {@link Pair#equals(Object)} for deep checking of arrays
35     */
36    @Test
37    public void testEquals() {
38      Pair<String, String> p1 = new Pair<String, String>("Hello", "World");
39      Pair<String, String> p1a = new Pair<String, String>("Hello", "World");
40      assertTrue(p1.equals(p1a));
41  
42      Pair<String, byte[]> p2 = new Pair<String, byte[]>("Hello", new byte[] { 1,
43          0, 5 });
44      Pair<String, byte[]> p2a = new Pair<String, byte[]>("Hello", new byte[] {
45          1, 0, 5 });
46      // Previously this test would fail as they are two different pointers to
47      // arrays that inherently the same.
48      assertTrue(p2.equals(p2a));
49  
50      Pair<char[], String> p3 = new Pair<char[], String>(new char[] { 'h', 'e' },
51          "world");
52      assertTrue(p3.equals(p3));
53  
54      // These kinds of tests will still fail as they have fundamentally different
55      // elements
56      Pair<Character[], String> p4 = new Pair<Character[], String>(
57          new Character[] { new Character('h'), new Character('e') }, "world");
58      // checking for autoboxing non-equality to the original class
59      assertFalse(p3.equals(p4));
60  
61      // still fail for a different autoboxing situation (just to prove that it
62      // is not just chars)
63      Pair<String, Integer[]> p5 = new Pair<String, Integer[]>("hello",
64          new Integer[] { new Integer(1), new Integer(982) });
65      Pair<String, int[]> p5a = new Pair<String, int[]>("hello", new int[] { 1,
66          982 });
67      assertFalse(p5.equals(p5a));
68  
69      // will still fail for that different things
70      Pair<String, byte[]> p6 = new Pair<String, byte[]>("Hello", new byte[] { 1,
71          0, 4 });
72      assertFalse(p2.equals(p6));
73  
74      // will still fail for the other predicate being different
75      Pair<String, byte[]> p7 = new Pair<String, byte[]>("World", new byte[] { 1,
76          0, 5 });
77      assertFalse(p2.equals(p7));
78    }
