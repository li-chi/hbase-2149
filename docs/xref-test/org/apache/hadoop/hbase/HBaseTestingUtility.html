205   public MiniDFSCluster startMiniDFSCluster(int servers, final File dir)
206   throws Exception {
207     // This does the following to home the minidfscluster
208     //     base_dir = new File(System.getProperty("test.build.data", "build/test/data"), "dfs/");
209     // Some tests also do this:
210     //  System.getProperty("test.cache.data", "build/test/cache");

214       this.clusterTestBuildDir = dir;

221     FileSystem fs = this.dfsCluster.getFileSystem();
222     this.conf.set("fs.defaultFS", fs.getUri().toString());
223     // Do old style too just to be safe.
224     this.conf.set("fs.default.name", fs.getUri().toString());
225     return this.dfsCluster;
226   }

306   /**
307    * Start up a minicluster of hbase, optionally dfs, and zookeeper.
308    * Modifies Configuration.  Homes the cluster data directory under a random
309    * subdirectory in a directory under System property test.build.data.
310    * Directory is cleaned up on exit.
311    * @param numMasters Number of masters to start up.  We'll start this many
312    * hbase masters.  If numMasters > 1, you can find the active/primary master
313    * with {@link MiniHBaseCluster#getMaster()}.
314    * @param numSlaves Number of slaves to start up.  We'll start this many
315    * datanodes and regionservers.  If numSlaves is > 1, then make sure
316    * hbase.regionserver.info.port is -1 (i.e. no ui per regionserver) otherwise
317    * bind errors.
318    * @throws Exception
319    * @see {@link #shutdownMiniCluster()}
320    * @return Mini hbase cluster instance created.
321    */
322   public MiniHBaseCluster startMiniCluster(final int numMasters,
323       final int numSlaves)
324   throws Exception {
325     LOG.info("Starting up minicluster with " + numMasters + " master(s) and " +
326         numSlaves + " regionserver(s) and datanode(s)");
327     // If we already put up a cluster, fail.
328     String testBuildPath = conf.get(TEST_DIRECTORY_KEY, null);
329     isRunningCluster(testBuildPath);
330     if (testBuildPath != null) {
331       LOG.info("Using passed path: " + testBuildPath);
332     }
333     // Make a new random dir to home everything in.  Set it as system property.
334     // minidfs reads home from system property.
335     this.clusterTestBuildDir = testBuildPath == null?
336       setupClusterTestBuildDir() : new File(testBuildPath);
337     System.setProperty(TEST_DIRECTORY_KEY, this.clusterTestBuildDir.getPath());
338     // Bring up mini dfs cluster. This spews a bunch of warnings about missing
339     // scheme. Complaints are 'Scheme is undefined for build/test/data/dfs/name1'.
340     startMiniDFSCluster(numSlaves, this.clusterTestBuildDir);
341     this.dfsCluster.waitClusterUp();
342 
343     // Start up a zk cluster.
344     if (this.zkCluster == null) {
345       startMiniZKCluster(this.clusterTestBuildDir);
346     }
347     return startMiniHBaseCluster(numMasters, numSlaves);
348   }

350   /**
351    * Starts up mini hbase cluster.  Usually used after call to
352    * {@link #startMiniCluster(int, int)} when doing stepped startup of clusters.
353    * Usually you won't want this.  You'll usually want {@link #startMiniCluster()}.
354    * @param numMasters
355    * @param numSlaves
356    * @return Reference to the hbase mini hbase cluster.
357    * @throws IOException
358    * @throws InterruptedException 
359    * @see {@link #startMiniCluster()}
360    */
361   public MiniHBaseCluster startMiniHBaseCluster(final int numMasters,
362       final int numSlaves)
363   throws IOException, InterruptedException {
364     // Now do the mini hbase cluster.  Set the hbase.rootdir in config.
365     createRootDir();
366     Configuration c = new Configuration(this.conf);
367     this.hbaseCluster = new MiniHBaseCluster(c, numMasters, numSlaves);
368     // Don't leave here till we've done a successful scan of the .META.
369     HTable t = new HTable(c, HConstants.META_TABLE_NAME);
370     ResultScanner s = t.getScanner(new Scan());
371     while (s.next() != null) {
372       continue;
373     }
374     LOG.info("Minicluster is up");
375     return this.hbaseCluster;
376   }

404   /**
405    * Stops mini hbase, zk, and hdfs clusters.
406    * @throws IOException
407    * @see {@link #startMiniCluster(int)}
408    */
409   public void shutdownMiniCluster() throws IOException {
410     LOG.info("Shutting down minicluster");
411     if (this.hbaseCluster != null) {
412       this.hbaseCluster.shutdown();
413       // Wait till hbase is down before going on to shutdown zk.
414       this.hbaseCluster.join();
415     }
416     if (!this.passedZkCluster) shutdownMiniZKCluster();
417     if (this.dfsCluster != null) {
418       // The below throws an exception per dn, AsynchronousCloseException.
419       this.dfsCluster.shutdown();
420     }
421     // Clean up our directory.
422     if (this.clusterTestBuildDir != null && this.clusterTestBuildDir.exists()) {
423       // Need to use deleteDirectory because File.delete required dir is empty.
424       if (!FSUtils.deleteDirectory(FileSystem.getLocal(this.conf),
425           new Path(this.clusterTestBuildDir.toString()))) {
426         LOG.warn("Failed delete of " + this.clusterTestBuildDir.toString());
427       }
428       this.clusterTestBuildDir = null;
429     }
430     LOG.info("Minicluster is down");
431   }

589    * Provide an existing table name to truncate
590    * @param tableName existing table
591    * @return HTable to that new table
592    * @throws IOException
593    */
594   public HTable truncateTable(byte [] tableName) throws IOException {
595     HTable table = new HTable(getConfiguration(), tableName);
596     Scan scan = new Scan();
597     ResultScanner resScan = table.getScanner(scan);
598     for(Result res : resScan) {
599       Delete del = new Delete(res.getRow());
600       table.delete(del);
601     }
602     resScan = table.getScanner(scan);
603     return table;
604   }

606   /**
607    * Load table with rows from 'aaa' to 'zzz'.
608    * @param t Table
609    * @param f Family
610    * @return Count of rows loaded.
611    * @throws IOException
612    */
613   public int loadTable(final HTable t, final byte[] f) throws IOException {
614     t.setAutoFlush(false);
615     byte[] k = new byte[3];
616     int rowCount = 0;
617     for (byte b1 = 'a'; b1 <= 'z'; b1++) {
618       for (byte b2 = 'a'; b2 <= 'z'; b2++) {
619         for (byte b3 = 'a'; b3 <= 'z'; b3++) {
620           k[0] = b1;
621           k[1] = b2;
622           k[2] = b3;
623           Put put = new Put(k);
624           put.add(f, null, k);
625           t.put(put);
626           rowCount++;
627         }
628       }
629     }
630     t.flushCommits();
631     return rowCount;
632   }
633   /**
634    * Load region with rows from 'aaa' to 'zzz'.
635    * @param r Region
636    * @param f Family
637    * @return Count of rows loaded.
638    * @throws IOException
639    */
640   public int loadRegion(final HRegion r, final byte[] f)
641   throws IOException {
642     byte[] k = new byte[3];
643     int rowCount = 0;
644     for (byte b1 = 'a'; b1 <= 'z'; b1++) {
645       for (byte b2 = 'a'; b2 <= 'z'; b2++) {
646         for (byte b3 = 'a'; b3 <= 'z'; b3++) {
647           k[0] = b1;
648           k[1] = b2;
649           k[2] = b3;
650           Put put = new Put(k);
651           put.add(f, null, k);
652           if (r.getLog() == null) put.setWriteToWAL(false);
653           r.put(put);
654           rowCount++;
655         }
656       }
657     }
658     return rowCount;
659   }

752   public int createMultiRegions(final Configuration c, final HTable table,
753       final byte[] columnFamily, byte [][] startKeys)
754   throws IOException {
755     Arrays.sort(startKeys, Bytes.BYTES_COMPARATOR);
756     HTable meta = new HTable(c, HConstants.META_TABLE_NAME);
757     HTableDescriptor htd = table.getTableDescriptor();
758     if(!htd.hasFamily(columnFamily)) {
759       HColumnDescriptor hcd = new HColumnDescriptor(columnFamily);
760       htd.addFamily(hcd);
761     }
762     // remove empty region - this is tricky as the mini cluster during the test
763     // setup already has the "<tablename>,,123456789" row with an empty start
764     // and end key. Adding the custom regions below adds those blindly,
765     // including the new start region from empty to "bbb". lg
766     List<byte[]> rows = getMetaTableRows(htd.getName());
767     List<HRegionInfo> newRegions = new ArrayList<HRegionInfo>(startKeys.length);
768     // add custom ones
769     int count = 0;
770     for (int i = 0; i < startKeys.length; i++) {
771       int j = (i + 1) % startKeys.length;
772       HRegionInfo hri = new HRegionInfo(table.getTableDescriptor(),
773         startKeys[i], startKeys[j]);
774       Put put = new Put(hri.getRegionName());
775       put.add(HConstants.CATALOG_FAMILY, HConstants.REGIONINFO_QUALIFIER,
776         Writables.getBytes(hri));
777       meta.put(put);
778       LOG.info("createMultiRegions: inserted " + hri.toString());
779       newRegions.add(hri);
780       count++;
781     }
782     // see comment above, remove "old" (or previous) single region
783     for (byte[] row : rows) {
784       LOG.info("createMultiRegions: deleting meta row -> " +
785         Bytes.toStringBinary(row));
786       meta.delete(new Delete(row));
787     }
788     // flush cache of regions
789     HConnection conn = table.getConnection();
790     conn.clearRegionCache();
791     // assign all the new regions IF table is enabled.
792     if (getHBaseAdmin().isTableEnabled(table.getTableName())) {
793       for(HRegionInfo hri : newRegions) {
794         hbaseCluster.getMaster().assignRegion(hri);
795       }
796     }
797     return count;
798   }
